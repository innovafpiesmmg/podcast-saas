        const navContainer = document.getElementById('nav-container');
        const contentContainer = document.getElementById('content-container');
        const rssInput = document.getElementById('rss-feed');

        let selectedService = 'spotify'; // Servicio por defecto

        // --- Renderizar Navegación ---
        function renderNav() {
            navContainer.innerHTML = ''; // Limpiar
            Object.keys(services).forEach(key => {
                const service = services[key];
                const button = document.createElement('button');
                button.className = `nav-button w-full flex items-center space-x-3 p-2 rounded-md text-gray-300 hover:bg-indigo-500 hover:text-white transition-colors duration-150`;
                if (key === selectedService) {
                    button.classList.add('active');
                }
                button.innerHTML = `${service.icon} <span>${service.name}</span>`;
                button.onclick = () => {
                    selectedService = key;
                    renderNav(); // Re-renderizar nav para estado activo
                    renderContent(); // Renderizar contenido
                };
                navContainer.appendChild(button);
            });
        }

        // --- Renderizar Contenido ---
        function renderContent() {
            const service = services[selectedService];
            if (!service) {
                contentContainer.innerHTML = '<p>Selecciona un servicio de la lista.</p>';
                return;
            }

            const rssValue = encodeURIComponent(rssInput.value);

            // Procesar los pasos para incluir enlaces dinámicos
            const stepsHtml = service.steps.map(step => {
                // Reemplaza [texto del enlace](url) con un <a>
                const linkMatch = step.match(/\[(.*?)\]\((.*?)\)/);
                if (linkMatch) {
                    let url = linkMatch[2];
                    // Reemplaza las variables en la URL
                    url = url.replace(/\$\{(.*?)\}/g, (match, p1) => {
                        try {
                            return eval(p1); // Evalúa variables como 'rssValue'
                        } catch (e) {
                            return '';
                        }
                    });
                    
                    const text = linkMatch[1];
                    const processedStep = step.replace(linkMatch[0], `<a href="${url}" target="_blank" class="text-indigo-600 font-medium hover:underline">${text}</a>`);
                    return `<li class="pb-2 mb-2 border-b border-gray-200">${processedStep}</li>`;
                }
                return `<li class="pb-2 mb-2 border-b border-gray-200">${step}</li>`;
            }).join('');
            
            // Renderizar nota informativa si existe
            const infoHtml = service.info ? `<div class="p-4 bg-blue-50 border-l-4 border-blue-400 text-blue-800 rounded-r-md mb-4">${service.info}</div>` : '';

            contentContainer.innerHTML = `
                <div class="flex items-center space-x-3 mb-4">
                    <span class="text-gray-900">${service.icon.replace('class="w-6 h-6"', 'class="w-10 h-10"')}</span>
                    <h2 class="text-2xl font-bold text-gray-900">${service.name}</h2>
                </div>
                ${infoHtml}
                <ol class="list-decimal list-inside space-y-3 text-gray-700">
                    ${stepsHtml}
                </ol>
            `;
        }

        // --- Inicialización ---
        document.addEventListener('DOMContentLoaded', () => {
            renderNav();
            renderContent();

            // Actualizar contenido si cambia el RSS (para los enlaces dinámicos)
            rssInput.addEventListener('input', renderContent);
        });

    </script>
